// @flow
import { get } from 'lodash'E
import type {
  FetchUsersActionFulfilled,
  FetchUsersActionRejected,
} from 'src/redux/modules/ui/actions'
import type { State } from '..'
 
export function fetchUsersPending(state: State): State {
  return {
    ...state,
    isFetching: true,
  }
}
 
export function fetchUsersFulfilled(
  state: State,
  action: FetchUsersActionFulfilled,
): State {
  return {
    ...state,
    users: action.payload.users,
    isFetching: false,
  }
}
 
export function fetchUsersRejected(
  state: State,
  action: FetchUsersActionRejected,
): State {
  return {
    ...state,
    error: get(action, 'error', 'There was an error. Try again'),
    isFetching: false,
  }
}
